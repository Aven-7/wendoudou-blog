el-form搜索筛选栏组件封装 ​

背景 ​

vue后台管理系统，会有很多表格页面搜索栏选项。 如果每一个功能表格的页面都要重复写相同的代码或者粘贴复制，想想都头大。

所以要考虑对表格页面搜索栏选项一个封装，统一配置引用，提升开发维护效率和界面统一。

完成后的效果大概就是长这样：

封装完整代码 ​

vue
<template>
  <div class="form-search">
    <el-form :size="size" inline>
      <!--  普通条件    -->
      <el-form-item v-for="item in searchForm" :key="item.prop" :label="item.label">
        <!-- 输入框 -->
        <el-input v-if="item.type==='Input'" v-model="searchData[item.prop]" size="mini" clearable
                  @clear="clearParams('searchData[item.prop]')" :style="item.cssStyle"/>
        <!-- 下拉框 -->
        <el-select v-if="item.type==='Select'" v-model="searchData[item.prop]" size="mini" filterable clearable
                   @change="item.change(searchData[item.prop])" :style="item.cssStyle">
          <el-option v-for="(op, index) in item.options" :key="index" :label="op.label" :value="op.value"/>
        </el-select>
        <!-- 单选 -->
        <el-radio-group v-if="item.type==='Radio'" v-model="searchData[item.prop]" :style="item.cssStyle">
          <el-radio v-for="ra in item.radios" :key="ra.value" :label="ra.value">{{ ra.label }}</el-radio>
        </el-radio-group>
        <!-- 单选按钮 -->
        <el-radio-group v-if="item.type==='RadioButton'" v-model="searchData[item.prop]"
                        @change="item.change && item.change(searchData[item.prop])" :style="item.cssStyle">
          <el-radio-button v-for="ra in item.radios" :key="ra.value" :label="ra.value">{{ ra.label }}</el-radio-button>
        </el-radio-group>
        <!-- 复选框 -->
<!--        <el-checkbox-group v-if="item.type==='Checkbox'" v-model="searchData[item.prop]" :style="item.cssStyle">-->
<!--          <el-checkbox v-for="ch in item.checkboxs" :key="ch.value" :label="ch.value">{{ ch.label }}</el-checkbox>-->
<!--        </el-checkbox-group>-->
        <!--  年份选择器      -->
        <el-date-picker
          :style="item.cssStyle"
          v-if="item.type==='year'"
          v-model="searchData[item.prop]"
          value-format="yyyy-MM-dd HH:mm:ss"
          type="year"
          placeholder="选择年"
          @change="item.change(searchData[item.prop])">
        </el-date-picker>
        <!-- 日期 -->
        <el-date-picker v-if="item.type==='Date'" v-model="searchData[item.prop]" :style="item.cssStyle"/>
        <!-- 时间 -->
        <el-time-select v-if="item.type==='Time'" v-model="searchData[item.prop]" type="" :style="item.cssStyle"/>
        <!-- 日期时间 -->
        <el-date-picker v-if="item.type==='DateTime'" v-model="searchData[item.prop]" type="datetime"
                        :disabled="item.disable && item.disable(searchData[item.prop])" :style="item.cssStyle"/>
        <!-- 日期和时间范围选择器  -->
        <el-date-picker
          :style="item.cssStyle"
          v-if="item.type==='daterange'"
          v-model="searchData[item.prop]"
          value-format="yyyy-MM-dd HH:mm:ss"
          type="daterange"
          start-placeholder="开始日期"
          end-placeholder="结束日期"
        />
        <!--  年月      -->
        <el-date-picker
          :style="item.cssStyle"
          v-if="item.type==='monthrange'"
          v-model="searchData[item.prop]"
          type="monthrange"
          range-separator="-"
          start-placeholder="开始月份"
          end-placeholder="结束月份">
        </el-date-picker>
        <!-- 日期和时间范围(最新)       -->
        <el-date-picker
          :style="item.cssStyle"
          v-if="item.type==='datetimerange'"
          v-model="searchData[item.prop]"
          value-format="yyyy-MM-dd HH:mm:ss"
          type="datetimerange"
          range-separator="至"
          start-placeholder="开始日期"
          end-placeholder="结束日期"
          @change="item.change(searchData[item.prop])"
        >
        </el-date-picker>
        <!-- 滑块 -->
        <el-slider v-if="item.type==='Slider'" v-model="searchData[item.prop]" :style="item.cssStyle"></el-slider>
        <!-- 开关 -->
        <el-switch v-if="item.type==='Switch'" v-model="searchData[item.prop]" :style="item.cssStyle"/>
      </el-form-item>

      <!--   更多条件   -->
      <el-form-item v-if="searchMoreForm.length > 0">
        <el-popover
          width="397"
          placement="bottom"
          trigger="click">
          <div class="more-filter">

            <el-form-item v-for="items in searchMoreForm" :key="items.prop" :label="items.label" label-position="top">
              <!-- 输入框 -->
              <el-input v-if="items.type==='Input'" v-model="searchData[items.prop]" size="mini" clearable
                        @clear="clearParams('searchData[items.prop]')" :style="{width: '100%'}"/>
              <!-- 下拉框 -->
              <el-select v-if="items.type==='Select'" v-model="searchData[items.prop]" size="mini" filterable clearable
                         @change="items.change(searchData[items.prop])" :style="{width: '100%'}">
                <el-option v-for="(ops, index_ops) in items.options" :key="index_ops" :label="ops.label" :value="ops.value"/>
              </el-select>
              <!-- 单选 -->
              <el-radio-group v-if="items.type==='Radio'" v-model="searchData[items.prop]" :style="{width: '100%'}">
                <el-radio v-for="ras in items.radios" :key="ras.value" :label="ras.value">{{ ras.label }}</el-radio>
              </el-radio-group>
              <!-- 单选按钮 -->
              <el-radio-group v-if="items.type==='RadioButton'" v-model="searchData[items.prop]"
                              @change="items.change && items.change(searchData[items.prop])" :style="{width: '100%'}">
                <el-radio-button v-for="ras in items.radios" :key="ras.value" :label="ras.value">{{ ras.label }}</el-radio-button>
              </el-radio-group>
              <!-- 复选框 -->
<!--              <el-checkbox-group v-if="items.type==='Checkbox'" v-model="searchData[items.prop]" :style="{width: '100%'}">-->
<!--                <el-checkbox v-for="chs in items.checkboxs" :key="chs.value" :label="chs.value">{{ chs.label }}</el-checkbox>-->
<!--              </el-checkbox-group>-->
              <!--  年份选择器      -->
              <el-date-picker
                :style="items.cssStyle"
                v-if="items.type==='year'"
                v-model="searchData[items.prop]"
                value-format="yyyy-MM-dd HH:mm:ss"
                type="year"
                placeholder="选择年"
                @change="items.change(searchData[items.prop])">
              </el-date-picker>
              <!-- 日期 -->
              <el-date-picker v-if="items.type==='Date'" v-model="searchData[items.prop]" :style="{width: '100%'}"/>
              <!-- 时间 -->
              <el-time-select v-if="items.type==='Time'" v-model="searchData[items.prop]" type=""
                              :style="{width: '100%'}"/>
              <!-- 日期时间 -->
              <el-date-picker v-if="items.type==='DateTime'" v-model="searchData[items.prop]" type="datetime"
                              :disabled="items.disable && items.disable(searchData[items.prop])" :style="items.cssStyle"/>
              <!-- 日期和时间范围选择器  -->
              <el-date-picker
                :style="{width: '100%'}"
                v-if="items.type==='daterange'"
                v-model="searchData[items.prop]"
                value-format="yyyy-MM-dd HH:mm:ss"
                type="daterange"
                start-placeholder="开始日期"
                end-placeholder="结束日期"
              />
              <!--  年月      -->
              <el-date-picker
                :style="{width: '100%'}"
                v-if="items.type==='monthrange'"
                v-model="searchData[items.prop]"
                type="monthrange"
                range-separator="-"
                start-placeholder="开始月份"
                end-placeholder="结束月份">
              </el-date-picker>
              <!-- 日期和时间范围(最新)       -->
              <el-date-picker
                v-if="items.type==='datetimerange'"
                :style="{width: '100%'}"
                :default-time="['00:00:00', '23:59:59']"
                v-model="searchData[items.prop]"
                value-format="yyyy-MM-dd HH:mm:ss"
                type="datetimerange"
                range-separator="至"
                start-placeholder="开始日期"
                end-placeholder="结束日期"
                @change="items.change(searchData[items.prop])"
              >
              </el-date-picker>
              <!-- 滑块 -->
              <el-slider v-if="items.type==='Slider'" v-model="searchData[items.prop]"
                         :style="{width: '100%'}"></el-slider>
              <!-- 开关 -->
              <el-switch v-if="items.type==='Switch'" v-model="searchData[items.prop]" :style="{width: '100%'}"/>
            </el-form-item>

          </div>
          <el-button
            slot="reference"
            type="text">
            <span>更多筛选</span>
            <i class="el-icon-arrow-down cs-pl-5"/>
          </el-button>
        </el-popover>
      </el-form-item>
    </el-form>

    <el-form v-if="isHandle" inline>
      <el-form-item v-for="item in searchHandle" :key="item.label">
        <el-button :type="item.type" :size="item.size || size" @click="item.handle()">
          {{ item.label }}
        </el-button>
      </el-form-item>
    </el-form>
  </div>
</template>

<script>

  <d2-container>
    <SctFormSearch
      class="searchMain"
      size="mini"
      :search-data="searchData"
      :search-form="searchForm"
      :search-more-form="searchMoreForm"
      :search-handle="searchHandle"/>
  </d2-container>
</template>

<script>

export default {
  name: 'wms-basic-supplier-index',
  components: {
    SctFormSearch: () => import('@/components/sct-form-search')
  },
  data() {
    return {
      // search表单查询参数
      searchData: {
        name: undefined,
        contact_person: undefined,
        create_time: undefined
      },
      // search表单外部条件查询参数结构
      searchForm: [
        {type: 'Input', label: '供应商名称', prop: 'name', placeholder: '请输入名称', cssStyle: {width: '200px'}},
        {type: 'Input', label: '联系人', prop: 'contact_person', placeholder: '请输入联系人', cssStyle: {width: '200px'}}
      ],
      // search表单更多条件查询参数结构
      searchMoreForm: [
        {type: 'Input', label: '供应商名称', prop: 'name', placeholder: '请输入名称', cssStyle: {width: '200px'}},
        {type: 'Input', label: '联系人', prop: 'contact_person', placeholder: '请输入联系人', cssStyle: {width: '200px'}},
        {type: 'datetimerange', label: '创建时间', prop: 'create_time', placeholder: '请选择创建时间'}
      ],
      // search表单按钮参数结构
      searchHandle: [
        {label: '查询', type: 'primary', size: 'mini', handle: this.getList},
        {label: '重置', type: '', size: 'mini', handle: this.resetForm},
      ]
    }
  },
  mounted() {

  },
  methods: {
    // search搜索表单重置
    resetForm() {
      Object.assign(this.$data.searchData, this.$options.data().searchData)
    },
    // 获取数据
    getList() {
      console.log(this.searchData, '312312')
    },

  }
}
</script>
